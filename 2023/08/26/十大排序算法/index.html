void counting_sort(int *ini_arr, int *sorted_arr, int n) 
{
    int *count_arr = (int *) malloc(sizeof(int) * 100);
    int i, j, k;
    for (k = 0; k < 100; k++)
        count_arr[k] = 0;
    for (i = 0; i < n; i++)
        count_arr[ini_arr[i]]++;
    for (k = 1; k < 100; k++)
        count_arr[k] += count_arr[k - 1];
    for (j = n; j > 0; j--)
        sorted_arr[--count_arr[ini_arr[j - 1]]] = ini_arr[j - 1];
    free(count_arr);
}

void radixsort(int *a, int n) 
{
  int i, b[MAX], m = a[0], exp = 1;

  for (i = 1; i < n; i++) 
  {
    if (a[i] > m) 
      m = a[i];
  }

  while (m / exp > 0) 
  {
    int bucket[BASE] = { 0 };

    for (i = 0; i < n; i++) 
    {
      bucket[(a[i] / exp) % BASE]++;
    }

    for (i = 1; i < BASE; i++) 
    {
      bucket[i] += bucket[i - 1];
    }

    for (i = n - 1; i >= 0; i--) 
    {
      b[--bucket[(a[i] / exp) % BASE]] = a[i];
    }

    for (i = 0; i < n; i++) 
    {
      a[i] = b[i];
    }

    exp *= BASE;

#ifdef SHOWPASS
    printf("\nPASS   : ");
    print(a, n);
#endif
  }
}
