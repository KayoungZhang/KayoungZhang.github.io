Qt常用控件之进度条QProgressBar

本文最后更新于：2023年7月3日 下午

QProgressBar 是一种横向或者纵向显示进度的控件，用来描述任务的完成情况。QProgressBar 继承于QWidget。

主要属性和方法

  • 获取和设置当前值
int value() const;
void setValue(int);

ui->progressBar->setValue(30);  //设置当前显示值
  • 获取和设置最大、最小值。如果最小值和最大值都设置为0，进度条会显示了一个繁忙指示，而不会显示当前的值。
int maximum() const;
void setMaximum(int);
int minimum() const;
void setMinimum(int);

ui->progressBar->setMinimum(0);  //设置最小值
ui->progressBar->setMaximum(100);  //设置最大值

  • 获取和设置滑动条的方向
Qt::Orientation orientation() const
void setOrientation(Qt::Orientation)

ui->progressBar->setOrientation(Qt::Horizontal);  //设置显示方向
  • 获取和设置进度条的文本是否显示
bool isTextVisible() const
void setTextVisible(bool visible)

ui->progressBar->setTextVisible(true);  //设置文字可见
  • 获取和设置显示文字的格式，“%p%”显示百分比，“%v”显示当前值，“%m”显示总步数。缺省为“%p%”
QString format() const
void setFormat(const QString &format)

ui->progressBar->setFormat("%p%");  //设置文字显示格式

信号

void valueChanged(int value)

实例

创建新工程，在UI设计界面，拖拽相应控件，修改控件名字，完成布局。

功能实现：

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
{
    ui->setupUi(this);

    /* 通过代码设置属性 */
    ui->progBar_download->setRange(0, 100);
    ui->progBar_download->setValue(0); //初始值为0
    ui->label_download->setText("");
    ui->label_copy->setText("");

    /* 创建download定时器，并设置时间间隔50ms */
    m_timer_download = new QTimer();
    m_timer_download->setInterval(50);
    /* 创建copy定时器，并设置时间间隔5ms */
    m_timer_copy = new QTimer();
    m_timer_copy->setInterval(5);

    /* 点击下载按钮开启计时 */
    connect(ui->btn_download, &QPushButton::clicked, this, [=](){
        ui->progBar_download->reset(); //便于重新开起下载
        m_timer_download->start();
        ui->label_download->setText("下载中...");
    });
    /* 定时时间间隔到，执行槽函数 */
    connect(m_timer_download, &QTimer::timeout, this, &Widget::timer1_timeout_slot);

    /* 点击拷贝按钮开启计时 */
    connect(ui->btn_copy, &QPushButton::clicked, this, [=](){
        ui->progBar_copy->reset(); //便于重新开起拷贝
        m_timer_copy->start();
        ui->label_copy->setText("拷贝中...");
    });
    /* 定时时间间隔到，执行槽函数 */
    connect(m_timer_copy, &QTimer::timeout, this, &Widget::timer2_timeout_slot);
}

Widget::~Widget()
{
    delete ui;
}

void Widget::timer1_timeout_slot()
{
    if(ui->progBar_download->value() == ui->progBar_download->maximum())
    {
        this->m_timer_download->stop();//下载完成，停止计时
        ui->label_download->setText("下载完成!");
    }
    else
    {
        ui->progBar_download->setValue(ui->progBar_download->value() + 1);//每隔50ms值加1
    }

}

void Widget::timer2_timeout_slot()
{
    if(ui->progBar_copy->value() == ui->progBar_copy->maximum())
    {
        this->m_timer_copy->stop();
        ui->label_copy->setText("拷贝完成!");
    }
    else
    {
        ui->progBar_copy->setValue(ui->progBar_copy->value() + 1);
    }
}
