void Widget::on_btn_getPosAndSize_clicked()
{
    /* 获取窗口位置信息 */
    QRect rect = this->geometry();
    /* 在输出打印当前窗口位置信息，注意添加头文件<QDebug> */
    qDebug() << "左上角: " << rect.topLeft();
    qDebug() << "右上角: " << rect.topRight();
    qDebug() << "左下角: " << rect.bottomLeft();
    qDebug() << "右下角: " << rect.bottomRight();
    qDebug() << "宽度: " << rect.width();
    qDebug() << "高度: " << rect.height();
}

/* 设置窗口大小 */
void Widget::on_btn_setSize_clicked()
{
    this->resize(450, 450);
}

/* 固定窗口大小 */
void Widget::on_btn_setFixSize_clicked()
{
    this->setFixedSize(500, 500);
}

/* 最小缩放400*400 */
void Widget::on_btn_setMinSize_clicked()
{
    this->setMinimumSize(400, 400);
}

/* 最大缩放600*600 */
void Widget::on_btn_setMaxSize_clicked()
{
    this->setMaximumSize(600, 600);
}
/* 移动窗口到具体位置 */
void Widget::on_btn_move_clicked()
{
    this->move(100, 100);
}
/* 设置窗口标题 */
void Widget::on_btn_setTitle_clicked()
{
    this->setWindowTitle("QWidget测试");
}
/* 设置窗口logo */
void Widget::on_btn_setLogo_clicked()
{
    this->setWindowIcon(QIcon(":/logo.png"));
}
