Qt常用控件之列表框QListWidget

本文最后更新于：2023年7月3日 下午

QListWidget用于显示结构简单的列表，相对于父类QListView，它是“简易版”的 QListView，创建和使用列表的方式更简单、门槛更低。

QListWidget 类的继承关系如下：

QListWidget -> QListView -> QAbstractItemView -> QAbstractScrollArea -> QFrame -> QWidget

属性和方法

  • 获取和设置显示模式（可以在属性窗口设置，也可以用以下代码实现）
QListView::ViewMode viewMode() const
void setViewMode(QListView::ViewMode mode)

QListView::ListMode  //列表模式
QListView::IconMode  //图标模式
  • 添加条目
void addItem(const QString &label)
void addItem(QListWidgetItem *item)
void addItems(const QStringList &labels)
    
void insertItem(int row, QListWidgetItem *item)
void insertItem(int row, const QString &label)
void insertItems(int row, const QStringList &labels)
  • 删除条目
/* 方法一 */
QListWidgetItem* item = ui->lwProvince->currentItem();
ui->lwProvince->removeItemWidget(item);
delete item;

/* 方法二 */
int row = ui->lwProvince->currentRow();
QListWidgetItem* item = ui->lwProvince->takeItem(row);
delete item;

void QListWidget::clear()// 清空整个列表
  • 信号

void currentRowChanged(int currentRow)
void currentTextChanged(const QString &currentText)
void itemActivated(QListWidgetItem *item)

功能实现：

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
{
    ui->setupUi(this);

    /* 默认选中列表 */
    ui->radioButton_list->setChecked(true);
    ui->listWidget->setViewMode(QListView::ListMode);

    /* radioBtn与listWidget对应 */
    connect(ui->radioButton_list, &QRadioButton::clicked, this, [=](){
       ui->listWidget->setViewMode(QListView::ListMode);//列表模式
    });
    connect(ui->radioButton_icon, &QRadioButton::clicked, this, [=](){
        ui->listWidget->setViewMode(QListView::IconMode);//图标模式
    });
}

void Widget::on_btn_add_clicked()
{
    int iconIndex = QRandomGenerator::global()->bounded(1, 6);//产生随机数1~5
    QString fileName = ":/pic/city" + QString::number(iconIndex) + ".png";
    QIcon icon(fileName);

    QString str = ui->lineEdit->text();

    QListWidgetItem *item = new QListWidgetItem(icon, str);
    ui->listWidget->addItem(item);
}


void Widget::on_btn_insert_clicked()
{
    int iconIndex = QRandomGenerator::global()->bounded(1, 6);//产生随机数1~5
    QString fileName = ":/pic/city" + QString::number(iconIndex) + ".png";
    QIcon icon(fileName);

    QString str = ui->lineEdit->text();
    int row = ui->listWidget->currentRow();

    QListWidgetItem *item = new QListWidgetItem(icon, str);
    ui->listWidget->insertItem(row, item);
}


void Widget::on_btn_del_clicked()
{
    /* 方法一 */
    QListWidgetItem *item = ui->listWidget->currentItem();
    ui->listWidget->removeItemWidget(item);
    delete item;
    /* 方法二 */
//    int row = ui->listWidget->currentRow();
//    QListWidgetItem *item = ui->listWidget->takeItem(row);
//    delete item;
}
