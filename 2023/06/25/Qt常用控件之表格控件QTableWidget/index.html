Qt常用控件之表格控件QTableWidget

本文最后更新于：2023年7月3日 晚上

QTableWidget是QT中的表格组件类,一般用来展示多行多列的数据。QTableWidget中的每一个单元格都是一个QTableWidgetItem对象。

QTableWidgetItem的常用方法

QString text() const
setText(const QString &text)		设置单元格中的文本。
QIcon icon() const
setIcon(const QIcon &icon)			给单元格添加图标。
QBrush background() const
setBackground(const QBrush &brush)	设置单元格的背景。
QFont font() const   
setFont(const QFont &font)			设置单元格中文本的字体。
QBrush foreground() const
setForeground(const QBrush &brush)	设置单元格中字体的颜色。
int textAlignment() const
setTextAlignment(int alignment)		设置单元格中文本的对齐方式。
QString toolTip() const    
setToolTip(const QString &toolTip)	给单元格设置提示信息。

QAbstractItemView::SelectedClicked	当单击已选中的单元格才能编辑
QAbstractItemView::EditKeyPressed	当在项目上按下编辑键时才能编辑。
QAbstractItemView::AnyKeyPressed	当在项目上按下任何键时才能编辑
QAbstractItemView::AllEditTriggers	包含以上所有编辑操作
  • 获取和设置表格选择方式

void setSelectionBehavior(QAbstractItemView::SelectionBehavior behavior)

QAbstractItemView::SelectItems    选中单个单元格
QAbstractItemView::SelectRows     选中一行
QAbstractItemView::SelectColumns  选中一列
  • 获取和设置表格选择模式

QAbstractItemView::ContiguousSelection  相邻选择，按shift键选中一个范围内的行

功能实现：

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
{
    ui->setupUi(this);

    /* 首先需要设置行列数 */
    ui->tableWidget->setColumnCount(4);
    ui->tableWidget->setRowCount(1);
    /* 添加行标签 */
    QStringList horizontalHeaderLabel;
    horizontalHeaderLabel << "姓名" << "性别" << "年龄" << "籍贯";
    ui->tableWidget->setHorizontalHeaderLabels(horizontalHeaderLabel);
    /* 添加初始化数据 */
    QTableWidgetItem *nameItem = new QTableWidgetItem("李彤");
    QTableWidgetItem *genderItem = new QTableWidgetItem("女");
    QTableWidgetItem *ageItem = new QTableWidgetItem(QString::number(20));
    QTableWidgetItem *provinceItem = new QTableWidgetItem("上海");

    ui->tableWidget->setItem(0,0,nameItem);
    ui->tableWidget->setItem(0,1,genderItem);
    ui->tableWidget->setItem(0,2,ageItem);
    ui->tableWidget->setItem(0,3,provinceItem);

    /* 调节行高和列宽 */
//    ui->tableWidget->resizeRowsToContents();
//    ui->tableWidget->resizeColumnsToContents();
    ui->tableWidget->verticalHeader()->setSectionResizeMode(QHeaderView::ResizeToContents);
    ui->tableWidget->horizontalHeader()->setSectionResizeMode(QHeaderView::Interactive);

    /* 行列表头标签设置 */
    connect(ui->chekBox_row, &QCheckBox::stateChanged, this, [=](int state){
        state = ui->chekBox_row->checkState(); //获得当前checkBox状态
        if(state == Qt::Unchecked)
        {
            ui->tableWidget->horizontalHeader()->hide();
        }
        if(state == Qt::Checked)
        {
            ui->tableWidget->horizontalHeader()->show();
        }
    });
    connect(ui->chekBox_column, &QCheckBox::stateChanged, this, [=](int state){
        state = ui->chekBox_column->checkState();
        if(state == Qt::Unchecked)
        {
            ui->tableWidget->verticalHeader()->hide();
        }
        if(state == Qt::Checked)
        {
            ui->tableWidget->verticalHeader()->show();
        }
    });
    /* 设置表格可编辑 */
    connect(ui->chekBox_editable, &QCheckBox::stateChanged, this, [=](int state){
        state = ui->chekBox_editable->checkState();
        if(state == Qt::Unchecked)
        {
            ui->tableWidget->setEditTriggers(QAbstractItemView::NoEditTriggers);
        }
        if(state == Qt::Checked)
        {
            ui->tableWidget->setEditTriggers(QAbstractItemView::DoubleClicked | QAbstractItemView::EditKeyPressed);
        }
    });
    /* 交替显示背景 */
    connect(ui->chekBox_color, &QCheckBox::stateChanged, this, [=](int state){
        state = ui->chekBox_color->checkState();
        if(state == Qt::Unchecked)
        {
            ui->tableWidget->setAlternatingRowColors(false);
        }
        if(state == Qt::Checked)
        {
            ui->tableWidget->setAlternatingRowColors(true);
        }
    });

    /* 选择方式 */
    connect(ui->radioBtn_cell, &QRadioButton::clicked, this, [=](){
        if(ui->radioBtn_cell->isChecked())
        {
            ui->tableWidget->setSelectionBehavior(QAbstractItemView::SelectItems);
        }
    });
    connect(ui->radioBtn_row, &QRadioButton::clicked, this, [=](){
        if(ui->radioBtn_row->isChecked())
        {
            ui->tableWidget->setSelectionBehavior(QAbstractItemView::SelectRows);
        }
    });
    connect(ui->radioBtn_column, &QRadioButton::clicked, this, [=](){
        if(ui->radioBtn_column->isChecked())
        {
            ui->tableWidget->setSelectionBehavior(QAbstractItemView::SelectColumns);
        }
    });

    /* 添加一行 */
    connect(ui->btn_addRow, &QPushButton::clicked, this, [=](){
        int rows = ui->tableWidget->rowCount();
        ui->tableWidget->setRowCount(rows + 1);
    });
    /* 添加一列 */
    connect(ui->btn_addColumn, &QPushButton::clicked, this, [=](){
        int columns = ui->tableWidget->columnCount();
        ui->tableWidget->setColumnCount(columns + 1);
    });
    /* 插入一行 */
    connect(ui->btn_insertRow, &QPushButton::clicked, this, [=](){
        int currentRow = ui->tableWidget->currentRow();
        ui->tableWidget->insertRow(currentRow);
    });
    /* 插入一列 */
    connect(ui->btn_insertColumn, &QPushButton::clicked, this, [=](){
        int currentColumn = ui->tableWidget->currentColumn();
        ui->tableWidget->insertColumn(currentColumn);
    });
    /* 删除一行 */
    connect(ui->btn_deleteRow, &QPushButton::clicked, this, [=](){
        int currentRow = ui->tableWidget->currentRow();
        ui->tableWidget->removeRow(currentRow);
    });
    /* 删除一列 */
    connect(ui->btn_deleteColumn, &QPushButton::clicked, this, [=](){
        int currentColumn = ui->tableWidget->currentColumn();
        ui->tableWidget->removeColumn(currentColumn);
    });
    /* 清空表格内容 */
    connect(ui->btn_clearContens, &QPushButton::clicked, this, [=](){
        ui->tableWidget->clearContents();
    });
    /* 清空表格 */
    connect(ui->btn_deleteAll, &QPushButton::clicked, this, [=](){
        ui->tableWidget->clear();
    });
}
