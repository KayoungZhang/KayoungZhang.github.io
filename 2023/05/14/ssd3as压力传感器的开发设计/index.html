SPI协议

此传感器可以SPI传输，也可I2C传输，这里主要使用gpio 模拟SPI进行传输。

网上关于SPI的讲解有很多，搜集了网上的资料，可参考SPI协议详解 和 spi时序图怎么分析，怎么看懂spi时序图，这两篇博客写的特别清楚，详细介绍了spi协议的具体内容。

  1. spi通过时钟极性（CPOL）和时钟相位（CPHA）控制两个spi设备进行数据传输，所以共有四种模式。
  2. spi协议适用一主多从的工作场景，通过片选引脚可控制不同的从设备。
  3. spi传输的实质是主从设备交换数据，也就是你发一个数据必然会收到一个数据，你要收一个数据也要发一个数据（一般是0x00）。
  4. spi进行数据传输时是从最高位MSB到最低位LSB。

SPI模式与时序

CPOL=0 表示SCLK在空闲时为低电平。

CPOL=1 表示SCLK在空闲时为高电平。

CPHA=0 数据是在第一个跳变沿的时候就会被采集。

CPHA=1 数据是在第二个跳变沿的时候才会被采集。

/**
  * =================================================
  * |    模式	   | CPOL(时钟极性)  |	CPHA（时钟相位）|
  * =================================================
  * |MODE0(常用) |       0	   |        0       |
  * -------------------------------------------------
  * |MODE1  	|       0	    |        1       |
  *	-------------------------------------------------
  * |MODE2  	|       1	    |        0       | 
  *	-------------------------------------------------
  * |MODE3(常用) |	   1	   |        1       |
  *	=================================================
  * MODE0：SCK空闲时为低电平，数据在SCK上升沿被采样，下降沿输出
　* MODE1：SCK空闲时为低电平，数据在SCK下降沿被采样，上升沿输出
　* MODE2：SCK空闲时为高电平，数据在SCK下降沿被采样，上升沿输出
　* MODE3：SCK空闲时为高电平，数据在SCK上升沿被采样，下降沿输出
	*/
	
#include "spi.h"
#include <intrins.h>

void Delay1us()		//@11.0592MHz
{
	_nop_();
}
/**
  * @brief	spi初始化
  * @param	None
  * @retval	None
  */
void spi_init(void)
{
	SPI_CLK = 0;
	SPI_SS = 1;
}
/**
  * @brief  gpio 模拟spi,SPI模式0,CPOL=0，CPHA=0, MSB first，上升沿采样
  * @param  send_byte: 发送的一字节数据
  * @retval	read_dat:读取的一字节数据
  */
uint8_t soft_spi_rw_mode0(uint8_t send_byte)
{
	uint8_t i, read_dat = 0;
	
	SPI_SS = 0; //选择从设备
	SPI_CLK = 0; //空闲SCLK为0
	Delay1us();	
	
	for( i = 0; i < 8; i++ )
	{	
		/* 接收数据 */
		SPI_CLK = 1;   //拉高时钟，上升沿采样
		read_dat <<= 1;  
		if(SPI_MISO) 
			read_dat++; //若从从机接收到高电平，数据自加一
		Delay1us();
		/* 发送数据 */
		SPI_CLK = 0; 
		if(send_byte & 0x80) //下降沿输出数据
			SPI_MOSI = 1;  
		else                    
			SPI_MOSI = 0;  
		send_byte <<= 1;
		Delay1us();	
	}
	SPI_SS = 1; //释放从设备
	
	return read_dat;
}

/**
  * @brief  gpio 模拟spi,SPI模式1,CPOL=0，CPHA=1, MSB first，下降沿采样
  * @param  send_byte: 发送的一字节数据
  * @retval	read_dat:读取的一字节数据
  */
uint8_t soft_spi_rw_mode1(uint8_t send_byte) 
{
	uint8_t i, read_dat = 0;
	
	SPI_SS = 0; //选择从设备
    SPI_CLK = 0; //空闲状态SCLK为0
	Delay1us();
	
	for(i = 0; i < 8; i++)   
	{
		/* 发送数据 */
		SPI_CLK = 1;     //上升沿输出数据
		if(send_byte & 0x80)
			SPI_MOSI = 1;  
		else      
			SPI_MOSI = 0;   
		send_byte <<= 1;    
		Delay1us();
		/* 接收数据 */
		SPI_CLK = 0;     //拉低时钟，下降沿采样
		read_dat <<= 1;     
		if(SPI_MISO)
			read_dat++;     //若从从机接收到高电平，数据自加一
		Delay1us();
	}
	SPI_SS = 1; //释放从设备
	
	return (read_dat);     
}

#define PRESSURE_GP_BASE  	2.42
#define PRESSURE_AP_BASE  	0

#define PRESSURE_DP_MIN		-2500
#define PRESSURE_GP_MAX		1000
#define PRESSURE_GP_MIN		0
#define PRESSURE_AP_MAX		120000UL
#define PRESSURE_AP_MIN		0

#include "ssd3as.h"

/**
	* @brief  读取传感器压力值和温度值。
	* @param  pressure: 读取的压力值。
	* @param  temperature: 读取的温度值。
	* @param  sensor_type: 压力传感器类型
	* @retval None
	* @attention 传感器的温度读出来有问题，应该跟传感器有关。
	*/
void ssd_get_value(float* pressure, float* temperature, sensor_type_e sensor_type)
{
	uint16_t buf[4] = {0};   //用于存放压力AD值和温度AD值
	uint16_t pressure_data = 0, temprature_data = 0;
	uint8_t i;
	
  /* readout value from sensor by spi */
	for(i = 0; i < 4; i++)  
	{
		buf[i] = soft_spi_rw_mode0(0x00);
	}
	
	pressure_data = (buf[0]<<8 | buf[1]) & 0x3fff; //14位压力值
	temprature_data = (buf[2]<<8 | buf[3]) & 0x7ff; //11位温度值	

	/* 计算压力值 */
	/* 1.A-type输出10%~90%    2.B-type输出5%~95%*/
	if(pressure_data > 16383*0.1 && pressure_data < 16383*0.9) 
	{
		switch(sensor_type)
		{
			case(SENSOR_AP):
			{
				*pressure = (pressure_data - 1638.3) * (PRESSURE_AP_MAX - PRESSURE_AP_MIN) / 13106.4 \
								+ PRESSURE_AP_MIN - PRESSURE_AP_BASE; 
				break;
			}
			case(SENSOR_GP):
			{
				*pressure = (pressure_data - 1638.3) * (PRESSURE_GP_MAX - PRESSURE_GP_MIN) / 13106.4 \
								+ PRESSURE_GP_MIN - PRESSURE_GP_BASE; 
				break;
			}
			case(SENSOR_DP):
			{
				*pressure = (pressure_data - 1638.3) * (PRESSURE_DP_MAX - PRESSURE_DP_MIN) / 13106.4 \
								+ PRESSURE_DP_MIN - PRESSURE_DP_BASE; 
				break;
			}
			default: break;
		}
	}  
	/* 计算温度值 */
	*temperature = temprature_data * 200.0 / 2047.0 - 50.0;
}
