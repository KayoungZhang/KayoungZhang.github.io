/**
  * @brief  ads1120 initial config
  * @param  sensor_type: ä¼ æ„Ÿå™¨ç±»å‹
  * @retval None
  */
void ads1120_confiuration(sensor_e sensor_type)
{
	/* ads1120 register configuration */
	uint8_t config[4] ={0, 0, 0, 0};

	switch (sensor_type)
	{	
		case PT100:
		{/* PT100 configuration */
			config[0] = ADS1120_MUX_0_1 | ADS1120_GAIN_8; /* register 0 configuration */
			config[1] = ADS1120_DR_20 | ADS1120_CC; /* register 1 configuration */
			config[2] = ADS1120_VREF_0 | ADS1120_REJECT_BOTH | ADS1120_IDAC_500; /* register 2 configuration */
			config[3] = ADS1120_IDAC1_AIN2 | ADS1120_IDAC2_AIN3; /* register 3 configuration */
			break;				
		}
		case K_TYPE:
		{/* K-type Thermocouple configuration */
			config[0] = ADS1120_MUX_0_1 | ADS1120_GAIN_32; /* register 0 configuration */
			config[1] = ADS1120_DR_20 | ADS1120_CC; /* register 1 configuration */
			config[2] = ADS1120_VREF_INT | ADS1120_REJECT_BOTH | ADS1120_IDAC_OFF; /* register 2 configuration */
			config[3] = ADS1120_IDAC1_OFF | ADS1120_IDAC2_OFF; /* register 3 configuration */	
			break;				
		}
		case INTER:
		{/* ads1120 internal temperature sensor configuration */
			config[0] = 0; /* register 0 configuration */
			config[1] = ADS1120_CC | ADS1120_TEMP_SENSOR; /* register 1 configuration */
			config[2] = 0; /* register 2 configuration */
			config[3] = 0; /* register 3 configuration */	
			break;				
		}
		default: break;
	}
	ADS1120_CS = 0;	
	soft_spi_rw_mode1(ADS1120_RESET); // reset ads1120
	Delay100ms();
	ads1120_wr_reg(ADS1120_0_REGISTER, 0x04, config); //config ads1120 register
	Delay100ms();
	soft_spi_rw_mode1(ADS1120_START); //start conversation
	ADS1120_CS = 1;
}

void ads1120_rd_reg(uint8_t addr, uint8_t num, uint8_t* p_data)

/**

  * @param  sensor_type: ä¼ æ„Ÿå™¨ç±»å‹
  * @retval None
  */
void ads1120_get_temperature(float* temperature, sensor_e sensor_type)
{
	long value = 0;
	float voltage = 0.0f;
	
	value = ads1120_rd_data();
	switch (sensor_type)
	{	
		case PT100:
		{/* PT100 */	
			voltage = (float)(value * 1.65f / 8 / 32768);
			*temperature = (voltage * 2000 - 100)/0.385f;//IDAC_500uA, PT100æ¯å‡é«˜1â„ƒï¼Œé˜»å€¼å¢åŠ 0.385 ohm
			break;				
		}
		case K_TYPE:
		{/* K-type thermocouple */	
//			uart1_send_byte(calibration_value >> 8);
//			uart1_send_byte(calibration_value);
			voltage = (float)(value * 62.5 / 32  + cold_compensation); //voltage=value*2.048f/32/32768*1000000;è½¬æ¢ä¸ºuV
			*temperature = (float)(voltage / 41.0);	//K-typeèµ›è´å…‹ç³»æ•°ï¼š41â„ƒ/uV
			break;				
		}			
		case INTER:
		{/* ads1120 internal temperature sensor 14bit left-data */	
			value = (value >> 2) & 0x3fff;
			if(value & 0x2000) //negative temp
			{
				value = (~(value - 1))&0x1fff;	
				*temperature =(float)(value * (-0.03125f));
			}
			else
			{
				*temperature =(float)(value * 0.03125f);//postive temp	
			}		
			break;				
		}
		default: break;
	}
	return;
}

void ads1120_get_average_temp(float* temperature, sensor_e sensor_type)
{
	uint8_t i;
	float sum = 0.0; //æ¸©åº¦å’Œ
	
	for(i = 0; i < 8; i++)
	{

}
