TFTLCD简介

TFT-LCD（thin film transistor-liquid crystal display）即薄膜晶体管液晶显示器。液晶显示屏的每一个像素上都设置有一个薄膜晶体管（TFT），每个像素都可以通过点脉冲直接控制。

常用的通信模式有6800模式和8080模式，本例程使用8080并口模式（简称80并口），8080接口有5条基本的控制线和多条数据线（8/9/16/18位）：

  • CS：TFTLCD 片选信号。

  • WR：向 TFTLCD 写入数据。

  • RD：从 TFTLCD 读取数据。

  • RST：硬复位 TFTLCD。

  • RS：命令/数据标志（0，读写命令；1，读写数据）。

  • D[15：0]：16 位双向数据线。

例如点操作流程：

  • 画点：设置坐标 → 写 GRAM 指令 → 写入颜色数据 → LCD 上显示颜色；
  • 读点：设置坐标 → 读 GRAM 指令 → 读取颜色数据 → 获取到对应点的颜色；

设置TFT液晶显示通常需要以下几个步骤：

  • 设置STM32F1与TFTLCD模块连接的IO口，将与TFTLCD模块相连的IO口进行初始化，以便驱动LCD，ILI9341 的 8080 通讯接口时序可以由 STM32 普通 I/O 接口进行模拟，但这样效率太低，STM32 提供了一种特别的控制方法——使用 FSMC 接口实现 8080接口时序。

  • 初始化TFTLCD模块，向LCD控制器写入一系列的设置值（比如RGB格式、LCD显示方向、伽马校准等），这部分代码一般LCD厂商会提供，初始化之后LCD才可以正常使用。

  • 通过函数将字符和数字显示到TFTLCD模块上，点操作只是一个点的处理，因此需要设计一个函数来多次使用这个步骤以实现字符和数字的显示。

FSMC 简介

FSMC（Flexible Static Memory Controller即灵活的静态存储控制器）是STM32系列采用的一种新型存储器扩展技术，能够连接同步、异步存储器和16位PC存储卡，STM32通过FSMC可以与SRAM、ROM、PSRAM、NOR Flash和NAND Flash等存储器的引脚直接相连。

STM32F1的FSMC内部框图

从上图我们可以看出，STM32 的 FSMC 将外部设备分为 3 类：NOR/PSRAM 设备、NAND设备、PC卡设备。他们共用地址、数据总线等信号，具有不同的片选(NE[4:1])以区分不同的设备。

  • NOR Flash/PSRAM 控制器 异步 传输时序模式：

STM32CubeMX设置

  • RCC设置外接HSE，时钟设置为72M
  • 设置PB0为GPIO输出模式，上拉，高速，默认低电平。（背光控制信号）
  • 选择FSMC，选中Bank1，片选为NE4（即Bank1的第4区），存储器类型选为LCD Interface，LCD寄存器选择（RS）设置为A10，数据宽度设为16位

  • 在FSMC配置菜单中，使能存储器写，根据LCD驱动芯片的数据参数设置地址建立时间、数据保存时间、总线周转阶段持续时间（设置为0），模式A

  • 输入工程名，选择工程路径（不要有中文），选择MDK-ARM V5；勾选Generated peripheral initialization as a pair of ‘.c/.h’ files per peripheral；点击GENERATE CODE，生成工程代码

lcd_dev_t lcddev = {
	.width = LCD_WIDTH,
	.height = LCD_HEIGHT,
	.dir = LCD_VERTICAL  /* 默认为竖屏 */
};

/**
 * @brief       LCD写数据
 * @param       data: 要写入的数据
 * @retval      无
 */
void lcd_wr_data(volatile uint16_t value)
{
    *( __IO uint16_t * )(FSMC_ADDR_DATA) = value;
}
/**
 * @brief       LCD写指令
 * @param       cmd: lcd指令
 * @retval      无
 */
void lcd_wr_cmd(volatile uint16_t cmd)
{
    *( __IO uint16_t * )(FSMC_ADDR_CMD) = cmd;
}

/**
 * @brief       LCD读数据
 * @param       无
 * @retval      读取到的数据
 */
uint16_t lcd_rd_data(void)
{
    return (*( __IO uint16_t * )(FSMC_ADDR_DATA));
}

/**
 * @brief       LCD写寄存器
 * @param       cmd:指令
 * @param       data:要写入的数据
 * @retval      无
 */
void lcd_write_reg(uint16_t cmd, uint16_t value)
{
    *( __IO uint16_t * )(FSMC_ADDR_CMD) = cmd;       /* 写入要写的指令 */
    *( __IO uint16_t * )(FSMC_ADDR_DATA) = value;    /* 写入数据 */
}

/**
 * @brief       初始化LCD
 * @note        该初始化函数可以初始化各种型号的LCD(详见本.c文件最前面的描述)
 * @param       无
 * @retval      无
 */
void lcd_init(void)
{
	  /* 获取屏幕id */
	  lcddev.lcd_id = lcd_read_id();
	  /* 显示之前必须lcd芯片初始化 */
    if (lcddev.lcd_id == LCD_7789)
    {
        lcd_ex_st7789_reginit();    /* 执行ST7789初始化 */
    }
    else if (lcddev.lcd_id == LCD_9341)
    {
        lcd_ex_ili9341_reginit();   /* 执行ILI9341初始化 */
    }
    else if (lcddev.lcd_id == LCD_5310)
    {
        lcd_ex_nt35310_reginit();   /* 执行NT35310初始化 */
    }
    else if (lcddev.lcd_id == LCD_5510)
    {
        lcd_ex_nt35510_reginit();   /* 执行NT35510初始化 */
    }
    else if (lcddev.lcd_id == LCD_1963)
    {
        lcd_ex_ssd1963_reginit();   /* 执行SSD1963初始化 */
        lcd_ssd_set_backlight(100); /* 背光设置为最亮 */
    }
    /* 设置屏幕大小、方向及扫描方向 */
	  lcd_set_scan_dir(L2R_U2D);
    /* 开背光 */
    LCD_BL(1);   
    /* 清屏 */		
    lcd_clear(WHITE);
}
/**
 * @brief       读取屏幕id
 * @param       None
 * @retval      屏幕id
 */
uint16_t lcd_read_id(void)
{
	  /* 尝试9341 ID的读取 */
    lcd_wr_cmd(LCD9341_READ_ID);
    lcddev.lcd_id = lcd_rd_data();  /* dummy read */
    lcddev.lcd_id = lcd_rd_data();  /* 读到0X00 */
    lcddev.lcd_id = lcd_rd_data();  /* 读取93 */
    lcddev.lcd_id <<= 8;
    lcddev.lcd_id |= lcd_rd_data(); /* 读取41 */

    if (lcddev.lcd_id != LCD_9341)    /* 不是 9341 , 尝试看看是不是 ST7789 */
    {
        lcd_wr_cmd(LCD7789_READ_ID);
        lcddev.lcd_id = lcd_rd_data();      /* dummy read */
        lcddev.lcd_id = lcd_rd_data();      /* 读到0X85 */
        lcddev.lcd_id = lcd_rd_data();      /* 读取0X85 */
        lcddev.lcd_id <<= 8;
        lcddev.lcd_id |= lcd_rd_data();     /* 读取0X52 */
        
        if (lcddev.lcd_id == LCD_8552)        /* 将8552的ID转换成7789 */
        {
            lcddev.lcd_id = LCD_7789;
        }

        if (lcddev.lcd_id != LCD_7789)        /* 也不是ST7789, 尝试是不是 NT35310 */
        {
            lcd_wr_cmd(LCD5310_READ_ID);
            lcddev.lcd_id = lcd_rd_data();  /* dummy read */
            lcddev.lcd_id = lcd_rd_data();  /* 读回0X01 */
            lcddev.lcd_id = lcd_rd_data();  /* 读回0X53 */
            lcddev.lcd_id <<= 8;
            lcddev.lcd_id |= lcd_rd_data(); /* 这里读回0X10 */

            if (lcddev.lcd_id != LCD_5310)    /* 也不是NT35310,尝试看看是不是NT35510 */
            {
                /* 发送秘钥（厂家提供,照搬即可） */
                lcd_write_reg(0xF000, 0x0055);
                lcd_write_reg(0xF001, 0x00AA);
                lcd_write_reg(0xF002, 0x0052);
                lcd_write_reg(0xF003, 0x0008);
                lcd_write_reg(0xF004, 0x0001);
                
                lcd_wr_cmd(LCD5510_READ_IDH);           /* 读取ID高8位 */
                lcddev.lcd_id = lcd_rd_data();      /* 读回0X55 */
                lcddev.lcd_id <<= 8;

                lcd_wr_cmd(LCD5510_READ_IDL);           /* 读取ID低8位 */
                lcddev.lcd_id |= lcd_rd_data();     /* 读回0X10 */
                HAL_Delay(5);
                
                if (lcddev.lcd_id != LCD_5510)        /* 也不是NT5510,尝试看看是不是SSD1963 */
                {
                    lcd_wr_cmd(LCD1963_READ_ID);
                    lcddev.lcd_id = lcd_rd_data();
                    lcddev.lcd_id = lcd_rd_data();  /* 读回0X57 */
                    lcddev.lcd_id <<= 8;
                    lcddev.lcd_id |= lcd_rd_data(); /* 读回0X61 */

                    if (lcddev.lcd_id == LCD_5761)lcddev.lcd_id = LCD_1963; /* SSD1963读回的ID是5761H,为方便区分,我们强制设置为1963 */
                }
            }
        }
    }
		
		return lcddev.lcd_id;
}
/**
 * @brief       设置LCD的自动扫描方向(对RGB屏无效)
 * @note
 *              9341/5310/5510/1963/7789等IC已经实际测试
 *              注意:其他函数可能会受到此函数设置的影响(尤其是9341),
 *              所以,一般设置为L2R_U2D即可,如果设置为其他扫描方式,可能导致显示不正常.
 *
 * @param       dir:0~7,代表8个方向
 * @retval      无
 */
void lcd_set_scan_dir(uint8_t dir)
{
    uint16_t regval = 0;
    uint16_t dirreg = 0;
    uint16_t temp;

    /* 横屏时，对1963不改变扫描方向！其他IC改变扫描方向！竖屏时1963改变方向，其他IC不改变扫描方向 */
    if ((lcddev.dir == LCD_HORIZONTAL && lcddev.lcd_id != LCD_1963) || (lcddev.dir == LCD_VERTICAL && lcddev.lcd_id == LCD_1963))
    {
        switch (dir)   /* 方向转换 */
        {
            case 0: dir = 6; break;
            case 1: dir = 7; break;
            case 2: dir = 4; break;
            case 3: dir = 5; break;
            case 4: dir = 1; break;
            case 5: dir = 0; break;
            case 6: dir = 3; break;
            case 7: dir = 2; break;
        }
    }

		/* 根据扫描方式 设置 0X36/0X3600 寄存器 bit 5,6,7 位的值 */
		switch (dir)
		{
				case L2R_U2D:/* 从左到右,从上到下 */
						regval |= (0 << 7) | (0 << 6) | (0 << 5);
						break;

				case L2R_D2U:/* 从左到右,从下到上 */
						regval |= (1 << 7) | (0 << 6) | (0 << 5);
						break;

				case R2L_U2D:/* 从右到左,从上到下 */
						regval |= (0 << 7) | (1 << 6) | (0 << 5);
						break;

				case R2L_D2U:/* 从右到左,从下到上 */
						regval |= (1 << 7) | (1 << 6) | (0 << 5);
						break;

				case U2D_L2R:/* 从上到下,从左到右 */
						regval |= (0 << 7) | (0 << 6) | (1 << 5);
						break;

				case U2D_R2L:/* 从上到下,从右到左 */
						regval |= (0 << 7) | (1 << 6) | (1 << 5);
						break;

				case D2U_L2R:/* 从下到上,从左到右 */
						regval |= (1 << 7) | (0 << 6) | (1 << 5);
						break;

				case D2U_R2L:/* 从下到上,从右到左 */
						regval |= (1 << 7) | (1 << 6) | (1 << 5);
						break;
		}

		dirreg = LCD_SCAN_DIR;  /* 对绝大部分驱动IC, 由0X36寄存器控制 */

    if (lcddev.lcd_id == LCD_5510)
    {
        dirreg = LCD5510_SCAN_DIR;    /* 对于5510, 和其他驱动ic的寄存器有差异 */
    }

     /* 9341 & 7789 要设置BGR位 */
    if (lcddev.lcd_id == LCD_9341 || lcddev.lcd_id == LCD_7789)
    {
        regval |= 0X08;
    }

		lcd_write_reg(dirreg, regval);

		if (lcddev.lcd_id != LCD_1963)   /* 1963不做坐标处理 */
		{
				if (regval & 0X20)
				{
						if (lcddev.width < lcddev.height)   /* 交换X,Y */
						{
								temp = lcddev.width;
								lcddev.width = lcddev.height;
								lcddev.height = temp;
						}
				}
				else
				{
						if (lcddev.width > lcddev.height)   /* 交换X,Y */
						{
								temp = lcddev.width;
								lcddev.width = lcddev.height;
								lcddev.height = temp;
						}
				}
		}

		/* 设置显示区域(开窗)大小 */
		if (lcddev.lcd_id == LCD_5510)
		{
				lcd_wr_cmd(LCD5510_ADDR_COLUMN);
				lcd_wr_data(0);
				lcd_wr_cmd(LCD5510_ADDR_COLUMN + 1);
				lcd_wr_data(0);
				lcd_wr_cmd(LCD5510_ADDR_COLUMN + 2);
				lcd_wr_data((lcddev.width - 1) >> 8);
				lcd_wr_cmd(LCD5510_ADDR_COLUMN + 3);
				lcd_wr_data((lcddev.width - 1) & 0XFF);
				lcd_wr_cmd(LCD5510_ADDR_PAGE);
				lcd_wr_data(0);
				lcd_wr_cmd(LCD5510_ADDR_PAGE + 1);
				lcd_wr_data(0);
				lcd_wr_cmd(LCD5510_ADDR_PAGE + 2);
				lcd_wr_data((lcddev.height - 1) >> 8);
				lcd_wr_cmd(LCD5510_ADDR_PAGE + 3);
				lcd_wr_data((lcddev.height - 1) & 0XFF);
		}
		else
		{
				lcd_wr_cmd(LCD_ADDR_COLUMN);
				lcd_wr_data(0);
				lcd_wr_data(0);
				lcd_wr_data((lcddev.width - 1) >> 8);
				lcd_wr_data((lcddev.width - 1) & 0XFF);
				lcd_wr_cmd(LCD_ADDR_PAGE);
				lcd_wr_data(0);
				lcd_wr_data(0);
				lcd_wr_data((lcddev.height - 1) >> 8);
				lcd_wr_data((lcddev.height - 1) & 0XFF);
		}
}
/**
 * @brief       清屏函数
 * @param       color: 要清屏的颜色
 * @retval      无
 */
void lcd_clear(uint16_t color)
{
    uint32_t index = 0;
    uint32_t totalpoint = lcddev.width;
    totalpoint *= lcddev.height;    /* 得到总点数 */
    lcd_set_cursor(0x00, 0x0000);   /* 设置光标位置 */
	  if(lcddev.lcd_id == LCD_5510)
		{
			  lcd_wr_cmd(LCD5510_WR_GRAM);        /* 开始写入GRAM */
		}
		else
		{
			  lcd_wr_cmd(LCD_WR_GRAM);        /* 开始写入GRAM */
		}

    for (index = 0; index < totalpoint; index++)
    {
			  lcd_wr_data(color);
    }
}

#include "./Lcd/tftlcd.h"
#include <stdio.h>
/* USER CODE END Includes */

/* USER CODE BEGIN 1 */
uint8_t x = 0;
uint8_t dev_id[12];
/* USER CODE END 1 */

/* USER CODE BEGIN 2 */
lcd_init();                                         /* 初始化LCD */
sprintf((char *)dev_id, "LCD ID:%04X", lcddev.lcd_id);  /* 将LCD ID打印到lcd_id数组 */
//	printf("LCD ID:%x\r\n", lcddev.lcd_id); /* 打印LCD ID 此处没有初始化串口，无法打印 */
/* USER CODE END 2 */

while (1)
{
    switch (x)
    {
        case 0: lcd_clear(WHITE);	break;
        case 1: lcd_clear(BLACK);	break;
        case 2: lcd_clear(BLUE);	break;
        case 3: lcd_clear(RED);		break;
        case 4: lcd_clear(MAGENTA);	break;
        case 5: lcd_clear(GREEN);	break;
        case 6: lcd_clear(CYAN);	break;
        case 7: lcd_clear(YELLOW);  break;
        case 8: lcd_clear(BRRED);	break;
        case 9: lcd_clear(GRAY);	break;
        case 10: lcd_clear(LGRAY);	break;
        case 11: lcd_clear(BROWN);	break;
    }

    lcd_show_string(10, 40, 240, 32, 32, "STM32", RED);
    lcd_show_string(10, 80, 240, 24, 24, "TFTLCD TEST", GREEN);
    lcd_show_string(10, 110, 240, 16, 16, "KayoungZhang", GREEN);
    lcd_show_string(10, 130, 240, 16, 16, (char *)dev_id, RED); /* 显示LCD ID */

    if (x++ == 12)
        x = 0;
    HAL_Delay(1000); //1s轮换一次画面
}
