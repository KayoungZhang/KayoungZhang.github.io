ds1302实时时钟芯片的开发

本文最后更新于：2023年5月6日 上午

DS1302芯片作为实时时钟使用，在开发过程中，使用比较频繁，时钟芯片PCF8563也有使用，这里主要写写DS1302这款芯片的开发设计。

特点

  • 电压：2.0V ~ 5.5V
  • 提供年、月、日、时、分、秒、星期等信息
  • 31 字节数据静态RAM
  • 通过简单的串行口与单片机进行通信

应用电路

引脚 功能
Vcc2 主电源
Vcc1 备用电源，如纽扣电池CR2032
X1、X2 外接32.768khz晶振
CE 控制信号
I/O 串行接口数据线
SCLK 串行接口时钟线

指令寄存器

BCD码

BCD码（Binary Coded Decimal‎），用4位二进制数来表示1位十进制数，所以4位二进制表示的数字最大到 9 。例：0001 0011表示13，1000 0101表示85，0001 1010不合法。在十六进制中的体现：0x13表示13，0x85表示85，0x1A不合法

BCD码转十进制：DEC=BCD/16*10+BCD%16; （2位BCD）

十进制转BCD码：BCD=DEC/10*16+DEC%10; （2位BCD）

DS1302写数据时需要将十进制数据转为BCD码的形式，读数据时需要将BCD码转为十进制数据形式。

通信时序

注意： 读时序时，发送完命令字的MSB位后，马上就读出数据，所以每个脉冲提前半拍。

程序实现

#ifndef DS1302_H
#define DS1302_H

#include <stc15.h>
typedef unsigned char uint8_t;

/* pin definition */
#define DS1302_IO		P34
#define DS1302_SCLK		P36
#define DS1302_CE		P35

/* register of rtc read address command */
#include REG_SECOND_READ   0x81
#include REG_MINUTE_READ   0x83
#include REG_HOUR_READ     0x85
#include REG_DATE_READ     0x87
#include REG_MONTH_READ    0x89
#include REG_DAY_READ      0x8b  //week
#include REG_YEAR_READ     0x8d
/* register of rtc write address command */
#include REG_SECOND_WRITE   0x80
#include REG_MINUTE_WRITE   0x82
#include REG_HOUR_WRITE     0x84
#include REG_DATE_WRITE     0x86
#include REG_MONTH_WRITE    0x88
#include REG_DAY_WRITE      0x8a
#include REG_YEAR_WRITE     0x8c
/* write protect */
#include REG_WR_PROTECT_READ    0x8f
#include REG_WR_PROTECT_WRITE   0x8e
#include WRITE_PROTECT_ENABLE   0x80
#include WRITE_PROTECT_DISABLE  0x00

/* 年 月 日 时 分 秒 周 */
typedef struct
{
	uint8_t year;
	uint8_t month;
	uint8_t date;
	uint8_t hour;
	uint8_t minute;
	uint8_t second;
	uint8_t day;	
}tm_t;

extern void ds1302_init(void);
extern void ds1302_read_time(tm_t* p_tm);
extern void ds1302_write_time(tm_t* p_tm);

#endif
/**
  ************************************************
  * @file     ds1302.c
  * @author   Kayoung
  * @version  V1.0
  * @date     20220327
  * @brief    ds1302时钟driver
  **********************************************
  */ 
#include "ds1302.h"

/* BCD转十进制DEC： DEC = BCD/16*10 + BCD%16;(2位BCD)
   十进制DEC转BCD： BCD = DEC/10*16 + DEC%10;(2位BCD)
   初始化时间：2022年03月27日17时20分50秒周七，存储格式是用BCD码 
*/
/**
  * @brief   ds1302初始化
  * @param   None
  * @retval  None
  */
void ds1302_init(void)
{
	DS1302_SCLK = 0;
	DS1302_CE = 0;
}

/**
  * @brief	  向ds1302写入一个字节
  * @param	  command: 命令字
  * @param    wr_data: 待写入的字节
  * @retval   None
  */
static void ds1302_wrbyte(uint8_t command, uint8_t wr_data)
{
	uint8_t i; //循环变量
	DS1302_CE = 1;
	/* 先写入command */
	for(i = 0; i < 8; i++)//LSB to MSB
	{
		DS1302_IO = command & (0x01<<i);
		DS1302_SCLK = 1; //51单片机速率慢，此处不需要延时，上升沿写入命令
		DS1302_SCLK = 0;
	}
	/* 再写入data */
	for(i = 0; i < 8; i++)//LSB to MSB
	{
		DS1302_IO = wr_data & (0x01<<i);
		DS1302_SCLK = 1; //51单片机速率慢，此处不需要延时,上升沿写入数据
		DS1302_SCLK = 0;
	}
	DS1302_CE = 0;
}

/**
  * @brief	 ds1302读出一个字节
  * @param	 command: 命令字
  * @retval  recv_byte: 接收的字节
  */
static uint8_t ds1302_rdbyte(uint8_t command)
{
	uint8_t recv_byte = 0;
	uint8_t i;//循环变量
	
	DS1302_CE = 1;
	/* 先发送命令 */
	for(i = 0; i < 8; i++)//LSB to MSB
	{
		DS1302_IO = command & (0x01 << i);
		DS1302_SCLK = 0;
		DS1302_SCLK = 1;//上升沿写入命令
	}
	for(i = 0; i < 7; i++)//LSB to MSB
	{		
		DS1302_SCLK = 1;//下降沿读出数据
		DS1302_SCLK = 0;
		if(DS1302_IO)
		{
			recv_byte |= (0x01 << i);
		}
	}

	DS1302_CE = 0;
	DS1302_IO = 0;  //解决读取时间是ff的问题
}

/**
  * @brief	 ds1302设置时间
  * @param	 p_tm: 时间结构体指针
  * @retval  None
  */
void ds1302_write_time(tm_t* p_tm)
{	
	ds1302_wrbyte(REG_WR_PROTECT_WRITE, WRITE_PROTECT_DISABLE);//关闭写保护
	/* 十进制转BCD码 */
	ds1302_wrbyte(REG_YEAR_WRITE, (p_tm->year / 10 * 16 + p_tm->year % 10));
	ds1302_wrbyte(REG_MONTH_WRITE, (p_tm->month / 10 * 16 + p_tm->month % 10));
	ds1302_wrbyte(REG_DATE_WRITE, (p_tm->date / 10 * 16 + p_tm->date % 10));
	ds1302_wrbyte(REG_HOUR_WRITE, (p_tm->hour / 10 * 16 + p_tm->hour % 10));
	ds1302_wrbyte(REG_MINUTE_WRITE, (p_tm->minute / 10 * 16 + p_tm->minute % 10));
	ds1302_wrbyte(REG_SECOND_WRITE, (p_tm->second / 10 * 16 + p_tm->second % 10));
	ds1302_wrbyte(REG_DAY_WRITE, (p_tm->day / 10 * 16 + p_tm->day % 10));
	
	ds1302_wrbyte(REG_WR_PROTECT_WRITE, WRITE_PROTECT_ENABLE);//打开写保护
}


/**
  * @brief	 ds1302读取时间
  * @param	 p_tm: 时间结构体指针
  * @retval  None
  */
void ds1302_read_time(tm_t* p_tm)
{
	/* BCD码转十进制 */
	p_tm->year = ds1302_rdbyte(REG_YEAR_WRITE) / 16 * 10 + ds1302_rdbyte(REG_YEAR_WRITE) % 16;
	p_tm->month = ds1302_rdbyte(REG_MONTH_WRITE) / 16 * 10 + ds1302_rdbyte(REG_MONTH_WRITE) % 16;
	p_tm->date = ds1302_rdbyte(REG_DATE_WRITE) / 16 * 10 + ds1302_rdbyte(REG_DATE_WRITE) % 16;
	p_tm->hour = ds1302_rdbyte(REG_HOUR_WRITE) / 16 * 10 + ds1302_rdbyte(REG_HOUR_WRITE) % 16;
	p_tm->minute = (REG_MINUTE_WRITE) / 16 * 10 + ds1302_rdbyte(REG_MINUTE_WRITE) % 16;
	p_tm->second = ds1302_rdbyte(REG_SECOND_WRITE) / 16 * 10 + ds1302_rdbyte(REG_SECOND_WRITE) % 16;
	p_tm->day = ds1302_rdbyte(REG_DAY_WRITE) / 16 * 10 + ds1302_rdbyte(REG_DAY_WRITE) % 16;
}
#include "stc15.h"
#include "ds1302.h"

/* 初始化时间：2022年03月27日17时20分50秒周七，存储格式是用BCD码 */
tm_t time = {22; 3; 27; 17; 20; 50; 7};
tm_t tm;
void Delay1000ms()		//@11.0592MHz
{
	unsigned char i, j, k;

	i = 43;
	j = 6;
	k = 203;
	do
	{
		do
		{
			while (--k);
		} while (--j);
	} while (--i);
}

void main(void)
{
	ds1302_init();
	/* set time */
	ds1302_write_time(&time); 
	while(1)
	{
		ds1302_read_time(&tm);
		TI = 1;
		printf("%d-%d-%d %d:%:%d 周%d \r\n", tm.year, tm.month, tm.date, tm.hour, tm.minute, tm.second, tm.day);
		while(!TI);
		TI = 0;
		Delay1000ms();
	}
}
