温度传感器DS18B20的应用

本文最后更新于：2023年5月11日 晚上

DS18B20温度传感器在日常应用中，使用比较广泛，以下简单介绍一下：

特点：

  • 单总线传输（1-Wire interface）
  • 供电范围： 3.0V ~ 5.5V
  • 测量温度范围：-55°C ~ +125°C
  • 精度：-10°C ~ +85°C（±0.5°C）
  • 分辨率： 9 ~ 12 bits
  • 每一个 DS1820 有唯一的系列号，因此多个 DS1820 可以存在于同一条单线总线上。

存储器图

单总线接口访问DS1820的协议

1.初始化 2.进行ROM功能操作 3.进行RAM功能操作 4.传输数据

ROM功能指令

ROM指令 功能 备注
Read ROM [33h] 读取DS18B20内部信息 总线存在多个DS18B20无法使用此命令
Match ROM [55h] 根据64位ROM信息匹配对应的DS18B20
Skip ROM [CCh] 跳过DS18B20 总线上只有一个DS18B20
Search ROM [F0h] 搜索DS18B20
Alarm Search [ECh] 搜索DS18B20 上一次碰到警报才搜索DS18B20

RAM功能指令

RAM命令 功能
Write Scratchpad [4Eh] 向暂存存储器写内容
Read Scratchpad [BEh] 读暂存存储器的内容
Copy Scratchpad [48h] 复制暂存存储器内容到非易失存储器 E2 RAM
Convert T [44h] 转换温度
Recall E2 [B8h] 重新调出 E2 RAM 存储到暂存存储器
Read Power Supply [B4h] 读供电方式

程序实现

#ifndef DEV_DS18B20_H
#define DEV_DS18B20_H

#include <stc15.h>
typedef unsigned char uint8_t;

#define DQ P37   //ds18b20接口定义

/******* ROM function command ****************/
#define	READ_ROM		(0x33)
#define	MATCH_ROM		(0x55)
#define	SKIP_ROM		(0xCC)
#define	SEARCH_ROM		(0xF0)
#define	ALARM_SEARCH	(0XEC)
/******* RAM function command ****************/
#define	WRITE_SCRATCHPAD	(0x4E)
#define	READ_SCRATCHPAD		(0xBE)
#define	COPY_SCRATCHPAD		(0x48) 	//把暂存器RAM中的数据复制到EEPROM中
#define	CONVERT_T			(0x44)
#define	RECALL_EE			(0xB8) 	//把EEPROM中的数据恢复到RAM中
#define	READ_POWER			(0xB4)
#define TH_LIMIT			(0x78)	//120℃
#define TL_LIMIT			(0x05)	//5℃
#define RESOLUTION_9		(0x1F)	//Thermometer Resolution Configuration
#define RESOLUTION_10		(0x3F)
#define RESOLUTION_11		(0x5F)
#define RESOLUTION_12		(0x7F)

extern uint8_t ds18b20_init(void);
extern void ds18b20_convert_temp(void);
extern float ds18b20_get_temp(void);

#endif
/**
  ************************************************
  * @file		dev_ds18b20.c
  * @author		Kayoung
  * @version	V1.0
  * @date		20220324
  * @brief		ds18b20测温度
  **********************************************
  */ 
#include "dev_ds18b20.h"
#include <intrins.h>

/**
  * @brief	检测温度传感器是否正常
  * @param	None 
  * @retval	1:失败 0：成功
*/
uint8_t ds18b20_init(void)
{
	uint8_t i; //延时循环变量
	uint8_t ack;//存储检测设备是否成功标志
	
	DQ = 1;
	DQ = 0;			
	i = 227;		//将总线拉低480us~960us，这里延时500us
	while (--i);
	
	DQ = 1;			//然后拉高总线，延时70us
	i = 29;
	while (--i);
	
	ack = DQ; //读取总线电平
	i = 227;	//延时500us
	while(i--);
	
	return ack;											
}
/**
  * @brief	ds18b20发送一个位
  * @param	sendbit：待发送的位
  * @retval	None
  */
void ds18b20_sendbit(uint8_t sendbit)
{
	uint8_t i; //延时时间循环变量
	
	DQ = 0; //主机拉低延时10us
	i = 2;
	while (--i); 
	
	DQ = sendbit; //发送位，延时50us
	i = 20;
	while (--i); 
	
	DQ = 1; //释放总线
}
/**
  * @brief	ds18b20接收到一位
  * @param	None
  * @retval	Bit: 从总线上接收到的一位
  */
uint8_t ds18b20_recvbit(void)
{
	uint8_t i;
	uint8_t Bit; //存储接收到的一位
	
	DQ = 0; //主机拉低，延时8us
	i = 1;
	while (--i);
	
	DQ = 1; //主机释放总线，延时8us
	i = 1;
	while (--i);
	
	Bit = DQ; //读取总线数据，延时50us
	i = 20;
	while (--i);
	
	return Bit;
}
/**
  * @brief	向温度传感器写一个字节
  * @param	wr_data：待写入的数据
  * @retval	None
*/
static void ds18b20_wr_byte(uint8_t wr_data)
{
	uint8_t j;

	for (j = 0; j < 8; j++)   		//From LSB to MSB
	{		
		ds18b20_sendbit(wr_data & (0x01 << j));
	}
}
/**
  * @brief	读取温度传感器一个字节
  * @param	None
  * @retval	recv_byte：读取的一个字节
*/
static uint8_t ds18b20_rd_byte(void)
{
	uint8_t recv_byte = 0; //存储接收到的数据
	uint8_t j;
		
	for (j = 0; j < 8; j++)			//from LSB to MSB
	{		
		if(ds18b20_recvbit())
		{
			recv_byte |= (0x01 << j);
		}
	}				
	return recv_byte;
}
/**
  * @brief	转换温度
  * @param	None
  * @retval	None
  */
void ds18b20_convert_temp(void)
{
	ds18b20_init();
	ds18b20_wr_byte(SKIP_ROM);  //跳过ROM
	ds18b20_wr_byte(CONVERT_T); //转换命令
}
/**
  * @brief	获取正负温度
  * @param	None
  * @retval	ftemp：温度值
  */
float ds18b20_get_temp(void)
{
	uint8_t TLSB, TMSB;//存储温度的两个字节
	int temp; //存储温度的整形
	float ftemp; //温度
	
	ds18b20_init();
	ds18b20_wr_byte(SKIP_ROM);
	ds18b20_wr_byte(READ_SCRATCHPAD);

	TLSB = ds18b20_rd_byte();
	TMSB = ds18b20_rd_byte();
	temp = (TMSB << 8) | TLSB; 
	ftemp = temp * 0.0625; //分辨率是0.0625
	
	return ftemp;//负数是以其补码形式存储的，所以此函数可以表示负温度
}

注意事项：

单总线传输过程中，有严格的延时时序要求，对于不同的MCU，延时时间会跟此程序不一致，需要根据其时序图进行相应的调整。此程序已上传Gitee：ds18b20
