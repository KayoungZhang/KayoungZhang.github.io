sht3x温湿度传感器的开发设计

本文最后更新于：2023年5月11日 晚上

国内市场碰到的温湿度传感器一般都采用的是瑞士公司Sensirion 生产的，像SHT2x 和 SHT3x和 SHTCx和 SHT4x等系列温湿度传感器。下面介绍一下其经典的数字温湿度传感器SHT31-DIS-B。

特点

  • 电源电压：2.4 - 5.5V

  • 相对湿度范围：0 - 100%RH 相对湿度精度：2%RH

  • 工作温度范围：-40 - 125°C 温度精度：0.2°C

  • 接口：I²C

  • 尺寸 (长x宽x高)：2.5 x 2.5 x 0.9 mm³

/* Generator polynomial for CRC8 */
#define POLYNOMIAL  0x31 // P(x) = x^8 + x^5 + x^4 + 1 = 00110001 ，省略了最高位1

/**
  * @brief	crc8计算
  * @param	crcdata：数据指针
  * @param  size：数据长度
  * @retval	None
  */
uint8_t crc8_calc(uint8_t *crcdata, uint8_t size)
{

	uint8_t Bit;        // bit mask
	uint8_t crc = 0xFF; // calculated checksum
	uint8_t byteCtr;    // byte counter
  
  // calculates 8-Bit checksum with given polynomial 
  for(byteCtr = 0; byteCtr < size; byteCtr++)
  {
    crc ^= (crcdata[byteCtr]);
    for(Bit = 8; Bit > 0; --Bit)
    {
      if(crc & 0x80) crc = (crc << 1) ^ POLYNOMIAL;
      else           crc = (crc << 1);
    }
  }
	return crc;
}

 * @brief  i2c start.

void i2c_init(void) 
{
	I2C_SDA = 1; 
	I2C_SCL = 1;

/**
 * @brief  i2c start. I2C_SCL、I2C_SDA同为高，I2C_SDA跳变成低之后，I2C_SCL跳变成低
 * @param  None.
 * @retval None.
 */
void i2c_start(void)
{
  	I2C_SDA = 1;
  	I2C_SCL = 1;
  	i2c_delay(10);
  	I2C_SDA = 0;
  	i2c_delay(10);
  	I2C_SCL = 0;
  	i2c_delay(10);
}
/**
 * @brief  i2c stop.I2C_SCL、I2C_SDA同为低，I2C_SCL跳变成高之后，I2C_SDA跳变成高
 * @param  None.
 * @retval None.
 */
void i2c_stop(void)
{
  	I2C_SDA = 0;
  	I2C_SCL = 0;
	i2c_delay(10);
  	I2C_SCL = 1;
  	i2c_delay(10);
  	I2C_SDA = 1;
  	i2c_delay(10);
}
/**
 * @brief  CPU从I2C总线设备发送8bit数据
 * @param  write_byte: 待写入的1字节数据
 * @retval res：(获取从机ack)ACK: 0, NACK：1.
 */
ack_e i2c_wr_byte(uint8_t write_byte)
{
	uint8_t i;
	ack_e ack_status;
	I2C_SCL = 0;
	for (i = 0; i < 8; i++)
	{//要发送8位，从最高位开始
		if (write_byte & 0x80) // I2C_SDA = (write_byte & 0x80) >> 7;	
		{
			I2C_SDA = 1;
		}
		else
		{
			I2C_SDA = 0;
		}	
		i2c_delay(10);
		I2C_SCL = 1;//输出I2C_SDA稳定后，拉高I2C_SCL给出上升沿，从机检测到后进行数据采样
		i2c_delay(50);
		I2C_SCL = 0;
		i2c_delay(10);
		write_byte <<= 1;
	}
	I2C_SDA = 1; //释放总线
	I2C_SCL = 1;
	i2c_delay(50);
	ack_status = I2C_SDA;
	I2C_SCL=0;
	
	return ack_status;
}

/**
 * @brief  CPU从I2C总线设备读取8bit数据
 * @param  recv_byte : 读取的1字节数据指针.
 * @param  ack_status：主机发送ack（ACK: 0, NACK：1.）
 * @retval None.
 */
void i2c_rd_byte(uint8_t* recv_byte, ack_e ack_status)
{
	uint8_t i;
	
	for (i = 0; i < 8; i++) //from MSB to LSB
	{
		*recv_byte <<= 1;
		I2C_SCL = 1;
		i2c_delay(30);	//延时等待信号稳定
		if(I2C_SDA == 1)
			(*recv_byte)++;
		I2C_SCL = 0; //fall edge for next value
		i2c_delay(10);
	}
	I2C_SDA = ack_status; 
	i2c_delay(10);
	I2C_SCL = 1;	//产生一个时钟
	i2c_delay(50);
	I2C_SCL = 0;
	i2c_delay(10);
	I2C_SDA = 1;//释放总线
}

#ifndef __SHT3X_H_
#define __SHT3X_H_

/* only the 7 MSBs of the I2C Read/Write header constitute the I2C Address. */
#define SHT3XADDR 0x44    //default address             
#define SHT3XWriteHeader SHT3XADDR<<1   //7 MSBs + write bit 0
#define SHT3XReadHeader  SHT3XWriteHeader|0x01   //7 MSBs + read bit 1

/* Error codes */
typedef enum{
  NO_ERROR       = 0x00, // no error
  ACK_ERROR      = 0x01, // no acknowledgment error
  CHECKSUM_ERROR = 0x02, // checksum mismatch error
  TIMEOUT_ERROR  = 0x04, // timeout error
  PARM_ERROR     = 0x80, // parameter out of range error
}etError;

/* Sensor Commands */
typedef enum{
  CMD_READ_SERIALNBR  = 0x3780, // read serial number
  CMD_READ_STATUS     = 0xF32D, // read status register
  CMD_CLEAR_STATUS    = 0x3041, // clear status register
  CMD_HEATER_ENABLE   = 0x306D, // enabled heater
  CMD_HEATER_DISABLE  = 0x3066, // disable heater
  CMD_SOFT_RESET      = 0x30A2, // soft reset
	
  CMD_MEAS_CLOCKSTR_H = 0x2C06, // measurement: clock stretching, high repeatability
  CMD_MEAS_CLOCKSTR_M = 0x2C0D, // measurement: clock stretching, medium repeatability
  CMD_MEAS_CLOCKSTR_L = 0x2C10, // measurement: clock stretching, low repeatability
	
  CMD_MEAS_POLLING_H  = 0x2400, // measurement: polling, high repeatability
  CMD_MEAS_POLLING_M  = 0x240B, // measurement: polling, medium repeatability
  CMD_MEAS_POLLING_L  = 0x2416, // measurement: polling, low repeatability
	
  CMD_MEAS_PERI_05_H  = 0x2032, // measurement: periodic 0.5 mps, high repeatability
//  CMD_MEAS_PERI_05_M  = 0x2024, // measurement: periodic 0.5 mps, medium repeatability

//  CMD_MEAS_PERI_1_M   = 0x2126, // measurement: periodic 1 mps, medium repeatability

void Delay1ms()		//@11.0592MHz
{
	unsigned char i, j;

	_nop_();
	i = 11;
	j = 190;
	do
	{
		while (--j);
	} while (--i);
}


/**
  * @brief	向sht3x传感器写指令
  * @param	command：指令
  * @retval	error（ack：0 or nack：1）
  */
static ack_e sht3x_wr_cmd(etCommands command)
{
  ack_e error; // ack or nack

  error  = i2c_wr_byte(command >> 8); //upper 8 bits
  error |= i2c_wr_byte(command & 0xFF); //lower 8 bits

  return error;
}

  * @brief	sht3x初始化
  * @param	None
  * @retval	None
  */
void sht3x_init(void)

	i2c_init();//I2C初始化	
}

/**
  * @brief	读取sht3x序列号
  * @param	serialNumber: sht3x序列号指针
  * @retval	error
  */
etError sht3x_rd_serial_num(unsigned long* serialNumber)
{
	etError error = NO_ERROR;           // error code
	uint8_t serialNumWords[6] = {0};      //serial number raw value and checksum value
  uint8_t i;

	i2c_start();
  error = i2c_wr_byte(SHT3XWriteHeader);	
	if(error == NO_ERROR)  error = sht3x_wr_cmd(CMD_READ_SERIALNBR); // write "read serial number" command
	Delay1ms();
	if(error == NO_ERROR)
	{
		i2c_start();
		error = i2c_wr_byte(SHT3XReadHeader);
	}

	/* read temperature and humidity raw values */
  if(error == NO_ERROR)
  {
		for(i = 0; i < 5; i++)
		{
			 i2c_rd_byte(&serialNumWords[i], ACK);
		}
    i2c_rd_byte(&serialNumWords[i], NACK);//i = 5
//		serialNumWords[6] = '\0';
		i2c_stop();
		if(serialNumWords[2] != crc8_calc(serialNumWords,2))     error = CHECKSUM_ERROR;
	  if(serialNumWords[5] != crc8_calc(&serialNumWords[3],2)) error = CHECKSUM_ERROR;	
  }
  /* calc serial number as 32-bit integer */
  if(error == NO_ERROR)
  {
		/* print serialNumber */
//		printf("serialNumber: 0x%02x%02x\r\n", serialNumWords[0], serialNumWords[1], serialNumWords[3], serialNumWords[4]);	

		*serialNumber = serialNumWords[0] << 8 | serialNumWords[1];
		*serialNumber <<= 16;
		*serialNumber |= serialNumWords[3] << 8 | serialNumWords[4];
	}
  return error;
}

/**
  * @brief	sht3x单次轮询模式获取温湿度数据
  * @param	temperature : 温度指针
  * @param	humidity: 湿度指针
  * @param  cmd：指令
  * @param  timeout：超时（ms）
  * @retval	error（ack：0 or nack：1）
  */
etError sht3x_single_polling(float* temperature, float* humidity, etCommands cmd, uint8_t timeout)
{
    etError error = NO_ERROR;           // error code
    uint16_t rawValueTemp, rawValueHumi;    // temperature and humidity raw value from sensor
    uint8_t Rdata[6] = {0};      //temperature and humidity raw value and checksum value
    uint8_t i;

    /* Start Measurement*/
    i2c_start();
    error = i2c_wr_byte(SHT3XWriteHeader);
    if(error == NO_ERROR)
    {
        error = sht3x_wr_cmd(cmd);		
    }
    i2c_stop();

    if(error == NO_ERROR)
    {
        i2c_start();
        error = i2c_wr_byte(SHT3XReadHeader);
    }
    while(timeout--)
    {
        i2c_stop();
        i2c_start();
        error = i2c_wr_byte(SHT3XReadHeader);	// check if the measurement has finished
        if(error == NO_ERROR) break; // if measurement has finished -> exit loop
        Delay1ms();
    }
    if(timeout == 0) error = TIMEOUT_ERROR;    // check for timeout error	

    /* read temperature and humidity raw values */
    if(error == NO_ERROR)
    {
        for(i=0; i<5; i++)
        {
            i2c_rd_byte(&Rdata[i], ACK);
        }
        i2c_rd_byte(&Rdata[i], NACK);
        i2c_stop();
        if(Rdata[2] != crc8_calc(Rdata,2))     error = CHECKSUM_ERROR;
        if(Rdata[5] != crc8_calc(&Rdata[3],2)) error = CHECKSUM_ERROR;	
    }
    /* calculate temperature in  and humidity in %RH */
    if(error == NO_ERROR)
    {
        rawValueTemp = (Rdata[0] << 8) | Rdata[1];
        rawValueHumi = (Rdata[3] << 8) | Rdata[4];
        *temperature = 175.0f * (float)rawValueTemp / 65535.0f - 45.0f;
        *humidity = 100.0f * (float)rawValueHumi / 65535.0f;
    }
    return error;
}

/**
  * @brief	sht3x 周期性获取温湿度数据
  * @param	temperature : 温度指针
  * @param	humidity: 湿度指针
  * @param  cmd：指令
  * @retval	error（ack：0 or nack：1）
  */
etError sht3x_periodic_mode(float* temperature, float* humidity, etCommands cmd)
{
    etError error = NO_ERROR;           // error code
    uint16_t rawValueTemp, rawValueHumi;    // temperature and humidity raw value from sensor
    uint8_t Rdata[6] = {0};      //temperature and humidity raw value and checksum value
    uint8_t i;		

    /* start measurement */
    i2c_start();
    error  = i2c_wr_byte(SHT3XWriteHeader);
    if(error == NO_ERROR) error = sht3x_wr_cmd(cmd);

    /* Readout of Measurement Results */
    if(error == NO_ERROR)
    {
        i2c_start();
        error  = i2c_wr_byte(SHT3XWriteHeader);
    }

    if(error == NO_ERROR) error = sht3x_wr_cmd(CMD_FETCH_DATA);
    if(error == NO_ERROR) 
    {
        i2c_start();
        error = i2c_wr_byte(SHT3XReadHeader);
    }
    /* read temperature and humidity raw values */
    if(error == NO_ERROR)
    {
        for(i=0; i<5; i++)
        {
            i2c_rd_byte(&Rdata[i], ACK);
        }
        i2c_rd_byte(&Rdata[i], NACK);
        i2c_stop();
        if(Rdata[2] != crc8_calc(Rdata,2))     error = CHECKSUM_ERROR;
        if(Rdata[5] != crc8_calc(&Rdata[3],2)) error = CHECKSUM_ERROR;	
    }
    /* calculate temperature in  and humidity in %RH */
    if(error == NO_ERROR)
    {
        rawValueTemp = (Rdata[0] << 8) | Rdata[1];
        rawValueHumi = (Rdata[3] << 8) | Rdata[4];
        *temperature = 175.0f * (float)rawValueTemp / 65535.0f - 45.0f;
        *humidity = 100.0f * (float)rawValueHumi / 65535.0f;
    }
    return error;
}
