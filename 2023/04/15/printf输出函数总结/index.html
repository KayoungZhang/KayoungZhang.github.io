printf输出函数总结

本文最后更新于：2023年10月16日 上午

在调试程序时，会经常需要将某些信息输出到电脑，便于查找问题原因，快速解决问题。printf函数作为已经封装好的信息输出函数，会被经常使用到，下面介绍一些它的使用方式：

C库函数-printf()：

函数声明原型：

int printf(const char * format, ...);

发送格式化输出到标准输出stdout。

返回值：正确返回输出的字符总数，错误返回负值。

  • format -- 即是格式化字符串，包含：（1）字符串常量；（2）格式控制字符串；（3）转义字符。

  • ... -- 可变参数，根据不同的 format 字符串，函数可能需要一系列的可变参数，每个参数包含了一个要被插入的值，替换了 format 参数中指定的每个 % 标签。参数的个数应与 % 标签的个数相同。

格式控制字符串形式:

%[flags][width][.prec][length]type  //%[标志][最小宽度][.精度][类型长度]类型
  • 标志
标志 含义
- 左对齐, 默认右对齐
0 右对齐时, 用0填充宽度.(默认用空格填充)
# 输出时加前缀0x
空格 输出值为正,在前加空格, 为负时加负号

例如：

printf("%5d\n",1000);     //输出:( 1000)默认右对齐,左边补空格
printf("%-5d\n",1000);     //输出:(1000 )左对齐,右边补空格
printf("%05d\n",1000);    //输出:(01000)前面补0
printf("%#x\n",1000);  //输出:0x3e8
printf("%.0f %#.0f\n",1000.0,1000.0);//输出:(1000 1000.)当小数点后不输出值时依然输出小数点
  • 类型长度
类型长度 修饰类型 含义
hh d,i,o,u,x char
h`` |d,i,o,u,x|short int| |l`|d,i,o,u,x|long int| |ll|d,i,o,u,x|long long int`

例如：

char a = 'a';
short int b = 123;
int  c = 123;
long int d = 123;
long long int e = 123;
printf("%hhd\n", a); // 97
printf("%hd\n", b); // 123
printf("%d\n", c); // 123
printf("%ld\n", d); // 123
printf("%lld\n", e); // 123
  • 类型
类型 含义
d 有符号10进制整型
p 地址
u 无符号10进制整型
o 无符号8进制整型
x or X 无符号16进制整型
c 字符
f 单、双精度浮点数(默认保留6位小数)
e or E 以指数形式输出单、双精度浮点数
g or G 以最短输出宽度,输出单、双精度浮点数
s 字符串

例如：

有符号十进制:  printf("%d",123);//输出123
无符号十进制:   printf("%u",-10);//不可以输出负数,输出429496786
无符号八进制:    printf("%o",123);//输出123
无符号十六进制： printf("0x%x 0x%X",123,123);  //输出0x7b 0x7B
无符号十六进制： printf("%#x %#X",123,123);  //输出0x7b 0X7B
字符型数据：  printf("%c",64);//输出A
字符串型数据： printf("%s","测试test");//输出：测试test
单精度浮点数： printf("%.9f",0.000000123);//输出0.000000123
科学计数法：  printf("%e %E",0.000000123,0.000000123);//输出1.230000e-07 1.230000E-07

C库中常用预定义宏

预定义宏 含义
__LINE__ 当前的函数行 (int)
__FILE__ 当前程序源文件 (char*)
__DATE__ 当前的编译日期
__TIME__ 当前编译时间
__STDC__ 遵循ANSI C标准时该标识被赋值为1
__FUNCTION__ 当前运行的函数 (char*)
__VA_ARGS__ C99标准中可变参数宏，目前似乎只有 gcc 支持
##__VA_ARGS__ 当可变参数的个数为0时，去掉多余 , ,防止编译出错

如：

printf("file: %s\n", __FILE__);
printf("function: %s\n", __FUNCTION__);
printf("line: %d\n", __LINE__);

# 字符串化操作符 与 ## 连接操作符

  • 字符串化操作符 #

可以使用 # 将当前的内容转换成字符串。如：

#include <stdio.h>
#define print(a) printf("%s = %d\n", #a, a);
int main(void)
{
    int x = 3;
    int y = 5;

    print(x / y);
    print(x + y);
    print(x * y);

    return 0;
}

/* 执行结果 */
x / y = 0
x + y = 8
x * y = 15
  • 连接操作符 ##

主要在编译的预处理阶段宏函数实现字符串连接的操作，常用的方式如下：

#define DEBUG(fmt, args…) printf(fmt, ##args)

armcc编译器器宏调试开关实例：

#include <stdio.h>

#define USE_DEBUG  //开启DEBUG宏 #undef DEBUG （不开启宏）

#if USE_DEBUG
#define DEBUG(fmt, args...) 	do{\
                              printf("file:%s function: %s line: %d " \
   						 								fmt, __FILE__, __FUNCTION__, __LINE__, ##args)\
   						 								}while(0)
#else
#define DEBUG(fmt, args...)

int main(int argc, char **argv) {
        char str[]="Hello World";
        DEBUG("%s",str);
        return 0;
}

gcc编译器宏调试开关可使用__VA_ARGS__可变参数宏：

#include <stdio.h>

#define USE_DEBUG  //开启DEBUG宏 #undef DEBUG （不开启宏）

#if USE_DEBUG
#define DEBUG(fmt, ...) 	do{\
                              printf("file:%s function: %s line: %d " \
   						 								fmt, __FILE__, __FUNCTION__, __LINE__, ##__VA_ARGS__)\
   						 								}while(0)
#else
#define DEBUG(fmt, ...)

int main(int argc, char **argv) {
        char str[]="Hello World";
        DEBUG("%s",str);
        return 0;
}

printf()重定向

在C语言中 printf 函数已经定义为默认输出到显示器的，在单片机中若要想在串口上显示，那就需要重定向 printf。例如将 printf 函数重定向到 USART1 ，这样就可以利用printf函数将单片机的数据通过串口调试助手打印到PC上。

printf函数底层实际是调用 fputc 函数，所以重定向 printf 函数，就是重新写 fputc 函数。

在 51单片机 中:

#include <stdio.h>
void uart1_send_byte(uint8_t value)
{
	SBUF = value;
	while (!TI); //等待发送完成
	TI = 0;
}
/* 重定向printf函数 */
char putchar(char ch)
{
	uart1_send_byte(ch);
	return ch;
}

在 STM32 中, 注意要勾选keil软件中 TARGET 中的 USE Micro LIB 选项, 然后进行重定义：

#include <stdio.h>

int fputc(int ch, FILE *f) 
{
    while ((USART1 -> SR & 0X40)== 0 );
    USART1 -> DR  = (uint8_t)ch;
    return  ch;
}
或者
int fputc(int ch, FILE *f) 
{
  USART_SendData(USARTx, (uint8_t)ch); /* 发送一个字节数据到串口 */
  while (USART_GetFlagStatus(USARTx, USART_FLAG_TXE) == RESET);/* 等待发送完毕 */
  return (ch);
}
或者
int fputc(int ch, FILE *f) 
{
  HAL_UART_Transmit(&huartx, (uint8_t *)&ch, 1, 0xffff);
  return ch;
}

此外，正点原子 有另外一种半主机模式，也可以实现printf输出，只做了解：

#pragma   import (__use_no_semihosting)
struct  __FILE
{
   int  handle ;
};

FILE  __stdout ;
_sys_exit (int  x)
{
   x  =  x ;
}

int  fputc (int  ch , FILE   * f)
{
   while ((USART1 -> SR & 0X40 )== 0);
    USART1 -> DR  = (uint8_t) ch ;
    return  ch ;
}
